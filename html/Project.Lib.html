Module Project.Lib

module Lib: sig .. end

Libraries.
type t 
val id : t -> string
Unique name of the library (to avoid name clashes with binaries having the same name).
val name : t -> string
The library name.
val filename : t -> string
The library filename. Usually, it is the same as name, but this could be updated when the library is put in a named project to project.name.
val units : t -> Project.Unit.t list
The list of compilation units which defines the library.
val available : t -> Project.Feature.formula
The features which enables the build of that library.
val deps : t -> Project.Dep.t list
The list of dependencies of the compilation units in the library.
val create : ?available:Project.Feature.formula ->
       ?flags:Project.Flags.t ->
       ?pack:bool ->
       ?deps:Project.Dep.t list -> Project.Unit.t list -> string -> t
Create a library.
val cma : t -> Project.Resolver.t -> string
The location of the generated bytecode archive.
val cmxa : t -> Project.Resolver.t -> string
The location of the extra information about the native archive.
val a : t -> Project.Resolver.t -> string
The location of the native archive.
val cmxs : t -> Project.Resolver.t -> string
The location of the shared archive.
val file : t -> Project.Resolver.t -> string -> string
file t ext is the generated file with the extension ext for the library t.
val generated_files : t ->
       Project.Resolver.t -> (Project.Feature.formula * string list) list
generated_files t r is the list of generated files when the given formula of features are enable.
val flags : t -> Project.Resolver.t -> Project.Flags.t
flags t r is the computed compilation flags, where r resolves a dependency names into a list of command-line flags.
val prereqs : t -> Project.Resolver.t -> [ `Byte | `Native ] -> string list
prereqs t resolver mode is the list of prerequisites files to build, in the given mode, before building the compilation unit t, where resolver is used to compute the location of generated files.
val build_dir : t -> Project.Resolver.t -> string
The build directory of the library.
