Module Project.Unit

module Unit: sig .. end

Compilation unit.
type t 
val id : t -> string
Unique name of the compilation unit. This is the composition of the build_path and the unit name.
val copy : t -> t
Copy the compilation unit.
val name : t -> string
The name of the compilation unit.
val dir : t -> string option
The source directory of the compilation unit.
val deps : t -> Project.Dep.t list
The dependencies of the compilation unit.
val container : t -> [ `Bin of Project.Bin.t | `Lib of Project.Lib.t ] option
The library the compilation unit belongs to.
val mli : t -> bool
Has the compilation unit an mli file.
val ml : t -> bool
val for_pack : t -> string option
Has the compilation unit an ml file.

The (optional) pack the compilation unit is in.

val create : ?flags:Project.Flags.t ->
       ?generated:[ `Both | `Ml | `Mli ] ->
       ?dir:string -> ?deps:Project.Dep.t list -> string -> t
Create a compilation unit.
val pack : ?flags:Project.Flags.t -> t list -> string -> t
Pack a collection of compilation units together.
val unpack : t -> t list
The (usually empty) list of packed compilation units.
val cmi : t -> Project.Resolver.t -> string
The location of the generated compiled module interface.
val cmo : t -> Project.Resolver.t -> string
The location of the generated compiled module object.
val cmx : t -> Project.Resolver.t -> string
The location of the extra information for native linking of the compilation unit.
val o : t -> Project.Resolver.t -> string
The location of the object file for the compilation unit.
val file : t -> Project.Resolver.t -> string -> string
file t ext is the generated file with the extension ext for the compilation unit t.
val generated_files : t ->
       Project.Resolver.t -> (Project.Feature.formula * string list) list
generated_files t r is the list of generated files when the given formula of features are enable.
val flags : t -> Project.Resolver.t -> Project.Flags.t
flags t r is the computed compilation flags, where r resolves a dependency names into a list of command-line flags.
val prereqs : t -> Project.Resolver.t -> [ `Byte | `Native ] -> string list
prereqs t resolver mode is the list of prerequisites files to build, in the given mode, before building the compilation unit t, where resolver is used to compute the location of generated files.
val build_dir : t -> Project.Resolver.t -> string
The build directory of the compilation unit.
